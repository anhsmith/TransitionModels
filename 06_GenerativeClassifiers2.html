Generative model both sexes

d23 <- d |> 
  filter(
    Length > 1050,
    Length < 1650,
    Population != "Northeast Pacific") |>
  mutate(Sex = Sex |> 
           fct_recode(Male = "M", Female = "F") |> 
           fct_relevel("Male", "Female"),
         Stage = Stage |> 
           fct_recode(Immature = "2", Mature = "3") 
         ) |> 
  droplevels()
